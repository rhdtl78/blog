import java.time.temporal.ChronoUnit;

import org.springframework.boot.convert.DurationUnit;
import org.springframework.stereotype.Component;

        this.name = name;
    }

    public int getAge() {
        return this.age;
    }

    public void setAge(int age) {
        this.age = age;
    }

}
